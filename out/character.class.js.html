/**
 * Represents a movable character in the game.
 * @extends MovableObject
 */

  width = 180;
  height = 290;
  y = 80;
  speed = 10;
  dateSet = false;

    "./img/2_character_pepe/5_dead/D-56.png",

  ];

  IMAGES_HURT = [
    "./img/2_character_pepe/4_hurt/H-41.png",
    "./img/2_character_pepe/4_hurt/H-42.png",
    "./img/2_character_pepe/4_hurt/H-43.png",
  ];

  walking_sounds = new Audio("audio/running.mp3");

    /**
   * @param {Object} world - The game world in which the character exists.
   */

  constructor() {
    super().loadImage("./img/2_character_pepe/2_walk/W-21.png");
    this.loadImages(this.IMAGES_WALKING);
    this.loadImages(this.IMAGES_JUMPING);
    this.loadImages(this.IMAGES_SLEEPING);
    this.loadImages(this.IMAGES_STANDING);
    this.loadImages(this.IMAGES_DEAD);
    this.loadImages(this.IMAGES_HURT);
    this.applyGravity();
    this.animate();
  }

  offset = {
    top: 130,
    left: 30,
    right: 40,
    bottom: 15,
  };

  isAnimating = true;

    /**
   * Animate the character's movements and actions.
   */
  animate() {
    setInterval(() => {
      this.walking_sounds.pause();
      if (this.world.keyboard.RIGHT && this.x < this.world.level.level_end_x) {
        this.walking_sounds.play();
        this.moveRight();
        this.otherDirection = false;
      }
      if (this.world.keyboard.LEFT && this.x > 0) {
        this.walking_sounds.play();
        this.moveLeft();
        this.otherDirection = true;
      }
      if (this.world.keyboard.SPACE && !this.isAboveGround()) {
        this.jump();
      }

      this.world.camera_x = -this.x + 100;
    }, 1000 / 60);

    setInterval(() => {
      if (this.isDead()) {
        this.playAnimation(this.IMAGES_DEAD);
        this.world.gameLost();
      } else if (this.isAboveGround()) {
        this.playAnimation(this.IMAGES_JUMPING);
        this.dateSet = false;
      } else if (this.isHurt()) {
        this.playAnimation(this.IMAGES_HURT);
        this.dateSet = false;
      } else if (this.world.keyboard.RIGHT || this.world.keyboard.LEFT) {
        this.playAnimation(this.IMAGES_WALKING);
        this.dateSet = false;
      } else if (
        this.dateSet === true &&
        (new Date().getTime() - this.time) / 2000 > 2
      ) {
        this.playAnimation(this.IMAGES_SLEEPING);
      } else {
        this.playAnimation(this.IMAGES_STANDING);
        if (this.dateSet === false) {
          this.time = new Date().getTime();
          this.dateSet = true;
        }
      }
    }, 170);
  }
}
